3.1 Standard ggplot2 colours

The standard ggplot2 colours are pretty nice. To understand these see The default colours of ggplot2.

To view the basic colours in a chart…

library(tidyverse)
library(ggpubr)

#create data frame
df <- data.frame(team=c('A', 'B', 'C'),
                 points=c(22, 28, 15))

#create bar plot using df
ggplot(df, aes(x=team, y=points, fill=team)) +
  geom_bar(stat = "identity")


To obtain the hex codes of these colours we can use the hue_pal function…

library(scales)

#extract hex color codes for a plot with three elements in ggplot2 
hex <- hue_pal()(3)

#display hex color codes
hex
## [1] "#F8766D" "#00BA38" "#619CFF"

show_col(hex)


To see the colours ggplot2 will use as standard with different numbers of groups…

#set margins of plot area
par(mai = c(0.1, 0, 0.1, 0), bg = "grey85")

#create plot with ggplot2 default colors from 1 to 8
gc.grid <- layout(matrix(1:8, nrow = 8))
for(i in 1:8){
   gc.ramp <- hue_pal()(i)
   plot(c(0, 8), c(0,1),
        type = "n", 
        bty="n", 
        xaxt="n", 
        yaxt="n", xlab="", ylab="")
   for(j in 1:i){
      rect(j - 1, 0, j - 0.25, 1, col = gc.ramp[j])
   }
}

3.2.1 Continuous scales

Colour gradients.. here illustrated with a 2d density estimate of the faithful dataset (Azzalini and Bowman 1990), which records the waiting time between eruptions and during each eruption for the Old Faithful geyser in Yellowstone Park.

  • eruptions numeric Eruption time in mins.
  • waiting numeric Waiting time to next eruption (in mins).
  • density 2d density estimate.

We hide the legends and set expand to 0, to focus on the appearance of the data.

Remember: although we use the erupt plot to illustrate concepts using with a fill aesthetic, the same ideas apply to colour scales. Any time we refer to scale_fill_() in this section there is a corresponding scale_colour_() for the colour aesthetic.

erupt <- ggplot(faithfuld, aes(waiting, eruptions, fill = density)) +
  geom_raster() +
  scale_x_continuous(NULL, expand = c(0, 0)) + 
  scale_y_continuous(NULL, expand = c(0, 0)) + 
  theme(legend.position = "none")

erupt
# scale_fill_gradient - produces a two-colour gradient
eruptb <- erupt + scale_fill_gradient(low = "#ff8200", high = "#7082d4")

# scale_fill_gradient2() produces a three-colour gradient with specified midpoint
eruptc <- erupt + scale_fill_gradient2(low = "#f25c91", mid = "white", high = "#00ab85", midpoint = mean(faithfuld$density))

# scale_fill_gradientn() produces an n-colour gradient
# Not sure of the use of this... 

ggarrange(erupt,
          eruptb,
          eruptc,
          labels = c("A", "B", "C"),
          ncol = 2, nrow = 2)

3.2.1.4 What is a good colour scale for continous data?

My preference for most charts is one colour from opaque to low opacity. As shown below, when you run from one colour to another the results are often hard to follow. The exception perhaps are maps, where a third middle colour, that can be white, works quite well. This does not work for something like a dot plot.

base <- ggplot(mpg, aes(cyl, displ, colour = hwy)) +
  geom_point(size = 2) 

basea <- base + scale_color_gradient(low = "#ccede8",high = "#00ab85")
baseb <- base + scale_color_gradient(low = "#f25c91",high = "#7082d4")
basec <- base + scale_color_gradient(low = "#F8766D",high = "#619CFF")
based <- base + scale_color_gradient2(low = "#F8766D",high = "#00BFC4", mid = "white", 
                                     midpoint = median(mpg$hwy))

ggarrange(
  basea, baseb, basec,based,
  labels = c("A", "B", "C","D"),
  ncol = 2, nrow = 2)

3.3 A how to guide on defining palettes

The below is taken from R for the rest of us.

Their first step is to define a list of colours as objects.

# Define Colors -----------------------------------------------------------

ga_purple <- "#8359AB"
ga_yellow <- "#FFDE39"
ga_gray <- "#827C78"
ga_blue <- "#49B8F1"
ga_brown <- "#B88262"
ga_pink <- "#DC458E"


They then put these in a function, so that these values are passed to the function scale_colour_manual each time it is run.

This first one is for categorical variables where the ordering of the colours is not meaningful.


# Qualitative -------------------------------------------------------------

scale_color_ga_qualitative <- function() {
  
  scale_color_manual(values = c(ga_blue, 
                                ga_pink, 
                                ga_yellow, 
                                ga_purple, 
                                ga_gray, 
                                ga_brown))
  
}

Next they do a sequential scale. Again, a function is defined where the progression of colour matters, setting this through the low and high arguments of the scale_fill_gradient function.

# Sequential --------------------------------------------------------------

scale_fill_ga_sequential <- function(low_color = ga_yellow, 
                                     high_color = ga_purple) {
  
  scale_fill_gradient(low = low_color, 
                      high = high_color)
  
}

nc_presidential_returns_by_county %>%

  geom_sf() +

To obtain diverging scale colours the scale_fill_gradient2 function allows a middle colour to be set.

# Diverging ---------------------------------------------------------------

scale_fill_ga_diverging <- function(low_color = ga_yellow, 
                                    medium_color = "white",
                                    high_color = ga_pink) {
  
  scale_fill_gradient2(low = low_color, 
                       mid = medium_color,
                       high = high_color)
  
}

nc_presidential_returns_by_county %>% 
  ggplot(aes(fill = per_point_diff)) +
  geom_sf() +
